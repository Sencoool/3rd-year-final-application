ฟังก์ชั่นในภาษา JavaScript

      ฟังก์ชัน (Function) คือ ชุดคำสั่งที่สามารถเรียกใช้ซ้ำได้ เพื่อช่วยให้โค้ดสั้นลง และง่ายต่อการจัดการ

1. การสร้างฟังก์ชันใน JavaScript

ใน JavaScript เราสามารถสร้างฟังก์ชันได้หลายแบบ เช่น

1.1 ฟังก์ชันแบบปกติ (Function Declaration)

ฟังก์ชันที่สามารถเรียกใช้งานได้ก่อนที่เราจะประกาศฟังก์ชัน

function sayHello() {
console.log("Hello, World!");
}

sayHello(); // เรียกใช้ฟังก์ชัน

1.2 ฟังก์ชันแบบนิพจน์ (Function Expression)

ฟังก์ชันที่ถูกเก็บไว้ในตัวแปร

const sayHi = function () {
console.log("Hi there!");
};

sayHi();

1.3 ฟังก์ชันลูกศร (Arrow Function - ES6)

เขียนโค้ดให้สั้นลง โดยใช้ => แทน function

const add = (a, b) => a + b;

console.log(add(5, 3)); // 8

2. ฟังก์ชันที่มีพารามิเตอร์และค่าที่ส่งกลับ (Return Value)

2.1 ฟังก์ชันที่รับพารามิเตอร์

function greet(name) {
console.log("Hello, " + name);
}

greet("Mart"); // Hello, Mart

2.2 ฟังก์ชันที่มี return

function multiply(a, b) {
return a * b;
}

let result = multiply(4, 5);
console.log(result); // 20

3. ค่าพารามิเตอร์เริ่มต้น (Default Parameters - ES6)

สามารถกำหนดค่าเริ่มต้นให้กับพารามิเตอร์ได้

function greet(name = "Guest") {
console.log("Hello, " + name);
}

greet(); // Hello, Guest
greet("Mart"); // Hello, Mart

4. ฟังก์ชันที่รับจำนวนพารามิเตอร์ไม่จำกัด (Rest Parameters)

ใช้ ... เพื่อรวมค่าพารามิเตอร์หลายค่าเป็น Array

function sum(...numbers) {
return numbers.reduce((total, num) => total + num, 0);
}

console.log(sum(1, 2, 3, 4)); // 10

5. ฟังก์ชันภายในฟังก์ชัน (Nested Function)

สามารถสร้างฟังก์ชันซ้อนกันได้

function outer() {
console.log("This is the outer function");

function inner() {
console.log("This is the inner function");
}

inner(); // เรียกใช้ฟังก์ชันที่อยู่ภายใน
}

outer();

6. ฟังก์ชันแบบ Callback (Callback Function)

ฟังก์ชันที่ถูกส่งเป็นพารามิเตอร์ไปยังฟังก์ชันอื่น

function processUserInput(callback) {
let name = "Mart";
callback(name);
}

processUserInput((name) => console.log("Hello, " + name));

7. ฟังก์ชันแบบ IIFE (Immediately Invoked Function Expression)

ฟังก์ชันที่ถูกเรียกใช้งานทันทีหลังจากที่ถูกประกาศ

(function () {
console.log("IIFE Function ถูกเรียกใช้งานทันที!");
})();

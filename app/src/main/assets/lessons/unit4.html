คำสั่งควบคุมในภาษา JavaScript

      Control Statements ในภาษา JavaScript คำสั่งควบคุมใน JavaScript ใช้สำหรับ กำหนดลำดับการทำงานของโปรแกรม สามารถแบ่งออกเป็น 3 ประเภท 1.คำสั่งเงื่อนไข (Conditional Statements) 2.คำสั่งวนซ้ำ (Loop Statements) 3.คำสั่งควบคุมการทำงานของลูป (Loop Control Statements)

1. คำสั่งเงื่อนไข (Conditional Statements)

1.1 if / else if / else

ใช้ตรวจสอบเงื่อนไข และเลือกทำงานตามเงื่อนไขที่เป็นจริง

let score = 75;
if (score >= 80) {
console.log("คุณได้เกรด A");
} else if (score >= 70) {
console.log("คุณได้เกรด B");
} else {
console.log("คุณได้เกรด C หรือต่ำกว่า");
}

1.2 Ternary Operator (? :)

เป็นรูปแบบย่อของ if-else ใช้เขียนเงื่อนไขสั้น ๆ

let age = 18;
let status = (age >= 18) ? "ผู้ใหญ่" : "เด็ก";
console.log(status); // ผู้ใหญ่

1.3 switch (ใช้แทน if-else หลายเงื่อนไข)

ใช้ตรวจสอบค่าของตัวแปรกับหลายเงื่อนไข

let day = "Monday";

switch (day) {
case "Monday":
console.log("วันนี้วันจันทร์");
break;
case "Tuesday":
console.log("วันนี้วันอังคาร");
break;
default:
console.log("วันนี้ไม่ใช่วันจันทร์หรืออังคาร");
}

2. คำสั่งวนซ้ำ (Loop Statements)

ใช้ให้โค้ดทำงานซ้ำ ๆ จนกว่าจะเป็นไปตามเงื่อนไขที่กำหนด

2.1 for loop (ใช้เมื่อรู้จำนวนรอบแน่นอน)

for (let i = 1; i <= 5; i++) {
console.log("รอบที่: " + i);
}

2.2 while loop (ใช้เมื่อไม่รู้จำนวนรอบแน่นอน)

let count = 1;
while (count <= 5) {
console.log("รอบที่: " + count);
count++; }

2.3 do...while loop (ทำงานก่อนเช็คเงื่อนไขเสมอ)

let num = 1;
do {
console.log("ค่าปัจจุบัน: " + num);
num++;
} while (num <= 5);

2.4 for...in (ใช้วนลูปผ่าน properties ของ Object)

let person = { name: "Mart", age: 25, city: "Bangkok" };

for (let key in person) {
console.log(key + ": " + person[key]);
}

2.5 for...of (ใช้วนลูปผ่านค่าของ Array หรือ iterable object)

let numbers = [10, 20, 30];

for (let num of numbers) {
console.log(num);
}

3. คำสั่งควบคุมการทำงานของลูป (Loop Control Statements)

ใช้สำหรับควบคุมการทำงานของลูป

3.1 break (ใช้หยุดลูปทันที)

for (let i = 1; i <= 5; i++) {
if (i === 3) break;
console.log(i);
}
// output: 1, 2

3.2 continue (ใช้ข้ามรอบปัจจุบัน แล้วไปทำรอบถัดไป)

for (let i = 1; i <= 5; i++) {
if (i === 3) continue;
console.log(i);
}
// output: 1, 2, 4, 5
}

4. คำสั่งจัดการข้อผิดพลาด (try...catch)

ใช้สำหรับ จัดการข้อผิดพลาด (Error Handling) ที่อาจเกิดขึ้นในโปรแกรม

try {
let result = 10 / 0;
console.log(result);
} catch (error) {
console.log("เกิดข้อผิดพลาด: " + error.message);
} finally {
console.log("โค้ดนี้ทำงานเสมอ");
}
